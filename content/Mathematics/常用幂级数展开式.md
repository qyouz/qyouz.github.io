$$
\begin{align} 
\frac{1}{1-x} &= \sum_{n=0}^{\infty} x^n, & x \in (-1, 1)  \\
\frac{1}{1+x} &= \sum_{n=0}^{\infty} (-1)^n x^n, & x \in (-1, 1) \\
\ln(1+x) &= \sum_{n=1}^{\infty} (-1)^{n-1} \frac{x^n}{n}, & x \in (-1, 1]  \\
\ln(1-x) &= -\sum_{n=1}^{\infty} \frac{x^n}{n}, & x \in [-1, 1)  \\
\frac{1}{2} \ln \frac{1+x}{1-x} &= \sum_{n=0}^{\infty} \frac{x^{2n+1}}{2n+1}, & x \in (-1, 1) \\
\frac{1}{1+x^2} &= \sum_{n=0}^{\infty} (-1)^n x^{2n}, & x \in (-1, 1)  \\
\arctan x &= \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n+1}}{2n+1}, & x \in [-1, 1]  \\
e^x &= \sum_{n=0}^{\infty} \frac{x^n}{n!}, & x \in (-\infty, +\infty) \\
\sin x &= \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n+1}}{(2n+1)!}, & x \in (-\infty, +\infty) \\
\cos x &= \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n}}{(2n)!}, & x \in (-\infty, +\infty) 
\end{align}
$$
